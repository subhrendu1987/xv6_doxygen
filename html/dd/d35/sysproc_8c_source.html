143 // if argint has trouble, return error -1

151 // create pointer to ptable processes
152 struct proc *ptr = ptable.proc;
153
154 // loop through ptable
155 for(; ptr < &ptable.proc[NPROC]; ptr++)
156 {
157 if(!(ptr->state == UNUSED))

165 }
166 }
